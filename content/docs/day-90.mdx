## ğŸ“‹ Contract Testing

Ensure service compatibility:

```javascript
// Pact contract test
const { Pact } = require("@pact-foundation/pact");

const provider = new Pact({
  consumer: "UserService",
  provider: "EmailService",
});

describe("Email Service Contract", () => {
  it("should send welcome email", () => {
    return provider
      .given("user registered")
      .uponReceiving("welcome email request")
      .withRequest({
        method: "POST",
        path: "/api/emails",
        body: {
          to: "user@example.com",
          subject: "Welcome!",
        },
      })
      .willRespondWith({
        status: 200,
        body: { sent: true },
      });
  });
});
```
